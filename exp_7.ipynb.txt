        "import numpy as np\n",
        "\n",
        "def initialize_centroids_random(X, k, random_state=None):\n",
        "    if random_state is not None:\n",
        "        np.random.seed(random_state)\n",
        "    indices = np.random.choice(X.shape[0], size=k, replace=False)\n",
        "    return X[indices].astype(float)\n",
        "\n",
        "\n",
        "def initialize_centroids_kmeanspp(X, k, random_state=None):\n",
        "    if random_state is not None:\n",
        "        np.random.seed(random_state)\n",
        "    n_samples = X.shape[0]\n",
        "    centroids = []\n",
        "    first_idx = np.random.randint(0, n_samples)\n",
        "    centroids.append(X[first_idx].astype(float))\n",
        "\n",
        "    for _ in range(1, k):\n",
        "        dists = np.array([min(np.sum((x - c) ** 2) for c in centroids) for x in X])\n",
        "        probs = dists / np.sum(dists)\n",
        "        next_idx = np.random.choice(n_samples, p=probs)\n",
        "        centroids.append(X[next_idx].astype(float))\n",
        "\n",
        "    return np.vstack(centroids)\n",
        "\n",
        "\n",
        "def assign_clusters(X, centroids):\n",
        "    dists = np.linalg.norm(X[:, None, :] - centroids[None, :, :], axis=2)\n",
        "    labels = np.argmin(dists, axis=1)\n",
        "    return labels\n",
        "\n",
        "\n",
        "def update_centroids(X, labels, k):\n",
        "    n_features = X.shape[1]\n",
        "    centroids = np.zeros((k, n_features))\n",
        "    for i in range(k):\n",
        "        members = X[labels == i]\n",
        "        if len(members) == 0:\n",
        "            centroids[i] = X[np.random.randint(0, X.shape[0])]\n",
        "        else:\n",
        "            centroids[i] = members.mean(axis=0)\n",
        "    return centroids\n",
        "\n",
        "\n",
        "def compute_inertia(X, centroids, labels):\n",
        "    ssd = 0.0\n",
        "    for i, c in enumerate(centroids):\n",
        "        members = X[labels == i]\n",
        "        if len(members) > 0:\n",
        "            ssd += np.sum((members - c) ** 2)\n",
        "    return ssd\n",
        "\n",
        "\n",
        "def kmeans(X, k, max_iters=100, tol=1e-4, init=\"kmeans++\", random_state=None, verbose=False):\n",
        "    if init == \"kmeans++\":\n",
        "        centroids = initialize_centroids_kmeanspp(X, k, random_state=random_state)\n",
        "    elif init == \"random\":\n",
        "        centroids = initialize_centroids_random(X, k, random_state=random_state)\n",
        "    else:\n",
        "        raise ValueError(\"init must be 'kmeans++' or 'random'\")\n",
        "\n",
        "    for it in range(1, max_iters + 1):\n",
        "        labels = assign_clusters(X, centroids)\n",
        "        new_centroids = update_centroids(X, labels, k)\n",
        "        shift = np.linalg.norm(new_centroids - centroids)\n",
        "\n",
        "        if verbose:\n",
        "            print(f\"Iter {it}: centroid shift {shift:.6f}\")\n",
        "\n",
        "        if shift < tol:\n",
        "            centroids = new_centroids\n",
        "            break\n",
        "\n",
        "        centroids = new_centroids\n",
        "\n",
        "    inertia = compute_inertia(X, centroids, labels)\n",
        "    return {\n",
        "        \"centroids\": centroids,\n",
        "        \"labels\": labels,\n",
        "        \"n_iter\": it,\n",
        "        \"inertia\": inertia\n",
        "    }\n"
